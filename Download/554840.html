#include<iostream>
#include<cstdio>
using namespace std;

int main(void)
{
  int i,cases;
  long long a,b,c;
  scanf("%d",&cases);
  for(i=1; i<=cases; ++i)
  {
    scanf("%lld %lld %lld",&a,&b,&c);
    if(a+b > c)
      printf("Case #%d: true\n",i);
    else
      printf("Case #%d: false\n",i);
  }
}
